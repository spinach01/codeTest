5. 한글 검사하기 

UTF8로 인코딩 된 문자열에서 한글초성으로만 된 문자가 포함된 경우 true를 그렇지 않으면 false를 출력한다.

예)
입력 : 하하ㅋㅋ
출력 : true

입력 : 안녕하세요
출력 : false

결과

코드

/*모든 문자는 UTF-8로 charset에 사전 지정*/
        
/*버튼 클릭시 실행. input에서 긁어서 find 함수에 넣어준다.*/
$("#resultBtn").on("click",function(){
    var inputArray = ($("#inputText").val()).trim().split("");
    $("#resultBox").html(find(inputArray));
})    

/*초성 모음 배열*/
var chosung = [ "ㄱ", "ㄲ", "ㄴ", "ㄷ", "ㄸ","ㄹ", "ㅁ", "ㅂ", "ㅃ", "ㅅ",
                "ㅆ", "ㅇ", "ㅈ", "ㅉ", "ㅊ","ㅋ", "ㅌ", "ㅍ", "ㅎ"       ];

/*초성을 비교해주는 함수*/
function find(inputArray){
    for(var i=0; i<inputArray.length; i++){
        var test = inputArray[i];
        for(var j=0; j<chosung.length; j++){
            if(chosung[j] == test){
                return "true";
                break;
            }
        }
    }
    return "false";
}
