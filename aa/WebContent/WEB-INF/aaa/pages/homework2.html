과제 내용

2. 미로를 만들어라.

입력된 숫자 N에 대하여 N*N 크기의 격자로 된 미로를 만들어라.
모든 격자를 통과하는 하나의 경로를 가지고 있어야 하며 입구는 (1,1)위쪽이며 출구는 (N,N)아래 이다.
격자 모서리는 +로 표기 하며 벽은 - 와 | 를 사용한다.
   
예)
입력 : 5
출력 : 
+ +-+-+-+-+
|         |
+-+-+-+-+ +
|         |
+ +-+-+-+-+
|         |
+-+-+-+-+ +
|         |
+ +-+-+-+-+
|         |
+-+-+-+-+ +

결과

코드

/*버튼 클릭시 실행. input에서 긁어서 코어 함수에 넣어준다.*/
$("#resultBtn").on("click",function(){
    $("tr").remove();
    var inputText = $("#inputText").val().trim();
    if(inputText!=""){
    makeTable(inputText);
    fillContent(inputText);
    }
}) 

/*사용자가 입력한 수에 따른 알맞는 테이블을 그려준다.*/
function makeTable(number){
    for(var i=0;i<number*2+1;i++){
        var tr = $("<tr>").appendTo("table");
        for(var j=0;j<number*2+1;j++){
            tr.append($("<td>").attr("id",i+"-"+j));  
        }
    }
}

/*패턴에 맞게 최소한의 반복으로 문자를 넣어준다.*/
function fillContent(number){
    for(var i=0;i<number*2+1;i++){
        for(var j=0;j<number*2+1;j++){
            if(i%2==0 && j%2==0){
                $("#"+i+"-"+j).html("+");
            }else if(i%2==0 && j%2==1){
                $("#"+i+"-"+j).html("-");
            }else if(i%2==1 && j==0 || i%2==1 && j==number*2){
                $("#"+i+"-"+j).html("|");
            } 
            if(i%4==0 && j==1 || i%4==2 && j==(number*2-1)){
                $("#"+i+"-"+j).html("");
            }
        }
    }
}
