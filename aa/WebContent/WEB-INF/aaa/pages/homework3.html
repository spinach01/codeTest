3. 숫자 배열 순서 바꾸기

입력된 숫자 배열을 다음과 같은 형태의 배열로 순서를 변경하라.

[a1, a2, a3..., an, b1, b2, b3..., bn]
=>
[a1, b1, a2, b2, a3, b3..., an, bn]

예)
입력 : 1 4 5 2 4 8
출력 : 1 2 4 4 5 8

결과

코드

/*버튼 클릭시 실행. input에서 긁어서 코어 reArrange 함수에 넣어준다.*/
$("#resultBtn").on("click",function(){
    var inputTextArray = $("#inputText").val().trim().split(" ");
    $("#resultBox").html(reArrange(inputTextArray).trim());
})  

/*숫자 배열이 2의 배수고 앞뒤로 정렬하므로 앞에서 하나 중간부터 하나를 붙여준다.
String 형에 담으면 문자열이 되고 Array 객체에 담으면 원하는 순서의 Array객체를 얻을 수 있다.*/
function reArrange(inputTextArray){
    var resultStr = "";
    for(var i=0;i<inputTextArray.length/2;i++){
        resultStr = resultStr+inputTextArray[i]+" ";
        resultStr = resultStr+inputTextArray[i+inputTextArray.length/2]+" ";
    }
    return resultStr;
}
