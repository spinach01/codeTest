4. 디지털 시계

디지털 시계에 하루동안(00:00~23:59) 3이 표시되는 시간을 초로 환산하면 총 몇 초(second) 인가?
디지털 시계는 하루동안 다음과 같이 시:분(00:00~23:59)으로 표시된다.

입력없이 결과(초)를 출력하면 된다.

/*버튼 클릭시 실행. timer함수를 실행시킨다.*/
$("#resultBtn").on("click",function(){  
    $("#resultBox").html("총 누적 초 :"+timer()));
})    

/*문자열에 숫자 3이 들어가면 true 안들어가면 false를 반환하는 함수.*/ 
function numberDetector(number){
    var numberArray = number.trim().split("");
    for(var count in numberArray){
        if(numberArray[count]=="3"){
            return true;
        }
    }
    return false;
}

/*시간과 분을 입력하면 초로 반환한는 함수.*/
function timeSecConvertor(hour,minute){
    return minute*60+hour*60*60;
}

/*시계 기능을 가지고 있는 함수.
60분씩 24시간을 돌면서 각 시간대에 3이 하나라도 포함되어 있는지 검사한다.
3이 포함되어 있는 시간을 초로 변환하여 누적한다.*/
function timer(){
    var totalSec=0;
    for(var i=0;i<24;i++){
        for(var j=0;j<60;j++){
            if(numberDetector(i+""+j)===true){
                totalSec=totalSec+timeSecConvertor(i,j);
            }
        }   
    }

    return totalSec;
}
