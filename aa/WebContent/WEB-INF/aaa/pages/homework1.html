1. 목표숫자에 맞게 연산 하기 

숫자 배열과 목표 숫자가 주어지면, 목표 숫자가 연산 결과로 나올 수 있도록 숫자 배열으로 이용한 연산 수식을 출력하라.
숫자 배열 사이에 연산자는 덧셈(+) 와 뺄셈(-)만 사용가능하며, 배열의 순서는 변경하지 않는다.

예) 
입력 : 1 3 4 5, 7        
1 - 3 + 4 + 5 = 7 이므로 
출력 : 1 - 3 + 4 + 5

코드

/*버튼 클릭시 실행. input에서 긁어서 코어 yunsan함수에 넣어준다.*/
$("#resultBtn").on("click",function(){
    var inputText = $("#inputText").val().trim().split(",");
    var inputNumberArray = inputText[0].trim().split(" ");
    var inputResult = inputText[1].trim();    
    $("#resultBox").html(yunsan(inputNumberArray,inputResult));
})    

/*이진수변환 값의 자리 수를 맞춰 주는 함수.*/
function jariAdjust(jariSu,num){
    if(num.toString().length<jariSu){
        var loopNum = jariSu-num.toString().length;
        for(var i=0;i<loopNum;i++){
            num="0"+num;
        }
    }
    return num;
}  

/*약속한 숫자를 부호로 바꿔주는 함수.*/
function numberToOperator(su){
    if(su==0){
        return "+";
    }else{
        return "-";
    }
}

/*실질적 연산을 하는 함수.
사용자가 입력한 숫자들의 배열과 결과 값을 입력 받고 필터 결과를 리턴한다.
중복되지 않고 부호가 가질 수 있는 확률 2^n(부호 수).
for문의 i값은 중복되지 않기 때문에 2진수로 변환해 부호로 적용.
-부호가 늘어나면 3진수 4진수.. 이렇게 늘리면 쉽게 적용가능.*/
function yunsan(inputNumberArray,resultNumber){
    var jariSu = (Math.pow(2, inputNumberArray.length-1)-1).toString(2).length;
    var resultStr;
        for(var i=0;i<Math.pow(2, inputNumberArray.length-1);i++){
            resultStr=""+inputNumberArray[0];
            var operatorJinsu = i.toString(2);
            operatorJinsu=jariAdjust(jariSu,operatorJinsu);
            for(var j=0;j<operatorJinsu.length;j++){
                resultStr = resultStr+""+numberToOperator(operatorJinsu[j]);
                resultStr = resultStr+""+inputNumberArray[j+1];
            }
            if(eval(resultStr)==resultNumber){
                return resultStr;
            }    
    }
    return "결과값을 잘 못 입력하셨습니다."
}
